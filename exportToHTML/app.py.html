from TheSmack.minilabs.minilab import Calculations

minilab_bp = Blueprint('minilab Menu', __name__,
                           template_folder='templates',
                           static_folder='static')

@minilab_bp.route('/student' , methods=['GET', 'POST'])
def student_minilab():
    if request.method == 'POST':
        num = int(request.form['num'])
        return render_template("/minilabs/minilab.html", calculations=Calculations(num))
    return render_template("/minilabs/minilab.html", calculations=Calculations.num)
